    INTERFACE
        CRYPTO_HW_ACCELERATOR
)

    target_include_directories(crypto_service_crypto_hw
        PUBLIC
            interface

    target_compile_definitions(crypto_service_crypto_hw
        PUBLIC
            CRYPTO_HW_ACCELERATOR
    )
endif()
